  <script>
    // Filter Scholarships
    function filterScholarships() {
      const course = document.getElementById('course-filter').value;
      const eligibility = document.getElementById('eligibility-filter').value;
      const scholarships = document.querySelectorAll('.scholarship');

      scholarships.forEach(sch => {
        const matches = (!course || sch.dataset.course === course) &&
                        (!eligibility || sch.dataset.eligibility === eligibility);
        sch.style.display = matches ? 'block' : 'none';
      });
    }

    // Show Apply Form
    function showApplyForm(scholarshipName) {
      document.getElementById('scholarship-name').value = scholarshipName;
      document.getElementById('apply-form').style.display = 'block';
      document.getElementById('success-message').style.display = 'none';
      document.getElementById('apply-form').scrollIntoView({ behavior: "smooth" });
    }

    // Submit Application
    document.getElementById('apply-form').addEventListener('submit', function(e) {
      e.preventDefault();
      const scholarship = document.getElementById('scholarship-name').value;
      let applied = JSON.parse(localStorage.getItem('appliedScholarships')) || [];
      applied.push({ name: scholarship, date: new Date().toLocaleDateString() });
      localStorage.setItem('appliedScholarships', JSON.stringify(applied));
      document.getElementById('apply-form').reset();
      document.getElementById('apply-form').style.display = 'none';
      document.getElementById('success-message').style.display = 'block';
      loadTracking();
    });

    // Load Application Tracking
    function loadTracking() {
      const applied = JSON.parse(localStorage.getItem('appliedScholarships')) || [];
      const list = document.getElementById('applied-list');
      list.innerHTML = '';
      applied.forEach(app => {
        const li = document.createElement('li');
        li.textContent = `${app.name} - Applied on ${app.date}`;
        list.appendChild(li);
      });
    }

    // Clear Applications
    function clearApplications() {
      localStorage.removeItem('appliedScholarships');
      loadTracking();
    }

    // Load on page ready
    window.onload = loadTracking;
  </script>
